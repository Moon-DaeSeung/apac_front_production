# TextField

## Basic usage

<Playground>
  {
    () => {
      const [value, setValue] = useState('')
      return(
          <TextField  value={value} onChange={setValue}/>
      )
    }
  }
</Playground>

## With error message

<Playground>
  {
    () => {
      const [value, setValue] = useState('')
      return(
          <TextField 
            label='7글자 미만 입력' 
            isError={value.length >= 7}
            value={value} 
            errorMessage='7글자 이상 입력했습니다'
            onChange={setValue}
          />
      )
    }
  }
</Playground>


## Customizing through Emotion css prop

<Playground>
  {
    () => {
      const [value, setValue] = useState('')
      return(
        <div style={{
          display: 'grid',
          width: '60%',
          gridTemplateColumns: 'repeat(2,1fr)', 
          gap: '10px', gridTemplateRows: 'repeate(3, 1fr)'
        }}>
          <TextField label='Default' value={value} onChange={setValue}/>
          <TextField label='Custom Size' customCss={css`width: 255px; height: 100px; font-size: 30px;`} value={value} onChange={setValue}/>
          <TextField label='Grid Attribute applied' customCss={css`width: auto; grid-column: 1 / 3;`} value={value} onChange={setValue}/>
          <TextField label='Etc Custom' customCss={css`border: dotted 2px black; background-color: yellow; color: black; font-size: 20px`} value={value} onChange={setValue}/>
         </div> 
      )
    }
  }
</Playground>
